# PHP-代码劫持

调用 ob_start() 来启用输出缓冲，这样，任何在 PHP 代码中产生的输出（例如 echo、print 等）都不会立即发送到浏览器，而是会被存储在缓冲区中

ob_start();
if(isset($_POST['c'])){
        $c= $_POST['c'];
        eval($c);
        $s = ob_get_contents();         
        ob_end_clean();
        echo preg_replace("/[0-9]|[a-z]/i","?",$s);
}else{
    highlight_file(__FILE__);
}
