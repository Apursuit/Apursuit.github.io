# 6. bash 比较注入

run_newsh.c 核心逻辑（以 root 权限调用 /opt/new.sh 并传参）：

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

int main(int argc, char *argv[]) {
    if (argc != 2) {
        fprintf(stderr, "Usage: %s <arg>\n", argv[0]);
        return 1;
    }
    setuid(0);
    setgid(0);
    char *script = "/opt/new.sh";
    char *args[] = { script, argv[1], NULL };
    execv(script, args);
    perror("execv failed");
    return 1;
}
